void switch(long baud) {
	bitTime = (1 / baud) * 1000000;
}


char sw_serial_getc(void) {
	char c;
	int i;
	start;//wait for the start bit before we start reading
	for(i = 0; i < 8; i++) {
		c |= rxPin;
		c << 1;
		delay_usec(bitTime);
	}
		
	return c;
}
